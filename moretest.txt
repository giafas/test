    task.wait(80)
    loadstring(game:HttpGet(('https://raw.githubusercontent.com/ReQiuYTPL/hub/main/ReQiuYTPLHub.lua'),true))()
    local playeraaaa, ailmentsFolderbbbb

    -- Wait until the player and ailmentsFolder exist
    repeat
        wait()
        playeraaaa = game.Players.LocalPlayer
        ailmentsFolderbbbb = playeraaaa and playeraaaa.PlayerGui and playeraaaa.PlayerGui.AilmentsMonitorApp and playeraaaa.PlayerGui.AilmentsMonitorApp.Ailments
    until playeraaaa and ailmentsFolderbbbb

    -- Function to check the number of ailments and kick the player if there are 5 or more
    local function checkAilmentscccc()
        if #ailmentsFolderbbbb:GetChildren() >= 5 then
            game.Players.LocalPlayer:Kick("Restart")
        end
    end

    -- Run the check initially
    checkAilmentscccc()

    -- Connect the function to the ChildAdded event
    ailmentsFolderbbbb.ChildAdded:Connect(checkAilmentscccc)
end)
